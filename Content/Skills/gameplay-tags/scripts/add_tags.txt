# add_tags.txt — Bulk-register gameplay tags and inspect them.
#
# Sample script for the gameplay-tags skill. Run via execute_python_code.
# NOTE: single-tag add/remove/rename and list-all now live in the engine GameplayTagsToolset
# (call_tool). VibeUE keeps bulk add_tags + has_tag / get_tag_info / get_children.
import unreal
gt = unreal.GameplayTagService

# Bulk-register several tags in one call (works for one or many)
r = gt.add_tags(
    ["Ability.Attack.Melee", "Ability.Attack.Ranged", "State.Stunned"],
    "added by sample script")
print("add ->", r.success, r.tags_modified)

print("has Ability.Attack.Melee:", gt.has_tag("Ability.Attack.Melee"))
print("children of Ability.Attack:",
      [t.tag_name for t in gt.get_children("Ability.Attack")])
